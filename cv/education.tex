%!TEX root = cv.tex

\section{\sc Education}

\begin{luacode*}

local utils = require 'utils'

tex.sprint(table.concat(utils.map(
    utils.load_db("../db/education.json"),
    function (e)
        return string.format("\\textbf{%s}, ", e.degree) ..
            table.concat(utils.map(e.majors, function(m) return string.format("{\\href{%s}{%s}}", m.url, m.name) end), ', ') ..
            string.format("\\hfill %s\\\\", e.date) ..
            string.format("{\\href{%s}{%s}}, %s", e.institute.url, e.institute.name, e.location)
    end
), "\\\\\\\\"))

\end{luacode*}
