function render_works(ws)
    local ws_strs = {}
    for _, w in ipairs(ws)
    do
        table.insert(ws_strs, string.format(
            "\\href{%s}{\\bf %s}, %s \\hfill %s\\\\%s",
            w.url, w.employer, w.location, w.date, w.title
        ))
    end
    return ws_strs
end

local utils = require 'utils'
tex.sprint(table.concat(render_works(utils.load_db("../db/work.json")), "\\\\\\\\"))

\end{luacode*}
